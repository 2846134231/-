#include<stdio.h>
int main ()
{
	char s[]="2+2*3+2/2-1";
	char x[100]={'\0'};
	int j,i,y=0;
	for(i=0,j=0;i<strlen(s);i++,j++)
	{
		if(s[i+1]=='*'||s[i+1]=='/')
		{
			if(s[i+1]=='*')
			{
				x[j]=(s[i]-'0')*(s[i+2]-'0')+'0';
			}
			if(s[i+1]=='/')
			{
				x[j]=(s[i]-0)/(s[i+2]-'0')+'0';
			}
			i+=2;
		}
		if(s[i+1]!='*'||s[i+1]!='/')
		{
			x[j]=s[i];
		}
		if(s[i+1]=='\0')
		{
			x[j]=s[i];
			x[j+1]=s[i+1];
		}
	}
	for(i=0;i<strlen(x);i++)
	{
		printf("%c\t",x[i]);
	}
	for(j=0;j<strlen(x);j++)
	{
		if(x[j+1]=='+')
		{
			y+=x[j]-'0';
		}
		if(x[j+1]=='-')
		{
			y=y-(x[j]-'0');
		}
		if(x[j+1]=='\0')
		{
			y+=x[j]-'0';
		}

	}
	printf("%d\n",y);
	return 0;
}
